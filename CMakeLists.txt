cmake_minimum_required (VERSION 3.8)
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED True)
project(Guccigedon VERSION 0.1)
add_executable(Guccigedon src/main.cpp)
find_package(Vulkan REQUIRED)
find_package(SDL2 )
if(NOT EXIST ${SDL2_INCLUDE_DIRS} )
    message("Cannot find SDL2")
    execute_process(COMMAND git submodule update --init -- vendor/SDL
                    WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR})
    # set(${SDL2_INCLUDE_DIRS} ${CMAKE_CURRENT_SOURCE_DIR}/vendor/SDL/include
    #     CACHE_PATH "SDL2 Include dir")
    # install(DIRECTORY ${SDL2_INCLUDE_DIRS})
    add_subdirectory(vendor/SDL)
endif()
include_directories(${Vulkan_INCLUDE_DIR} ${SDL2_INCLUDE_DIRS})
target_include_directories(Guccigedon PUBLIC)
target_link_libraries(Guccigedon ${SDL2_LIBRARIES} ${Vulkan_LIBRARY})
set (CMAKE_RUNTIME_OUTPUT_DIRECTORY "${PROJECT_SOURCE_DIR}/bin")
